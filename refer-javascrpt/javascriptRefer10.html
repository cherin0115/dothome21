01.요소객체 : 클래스 메서드 : .classList.add()/.classList.remove()

클릭하면 모든 이미지에 클래스(active)를 추가합니다. 클릭하면 모든 이미지에 클래스(active)를 삭제합니다. 
{
    document.querySelector(".btn1-1").addEventListener("click",function(){
        document.querySelectorAll(".ex1 div").forEach((el)=>{
            el.classList.add("active");
        });
    });

    document.querySelectorAll(".ex1 div").forEach((el)=>{
        el.classList.remove("active");
        });
    });
}

02. 요소객체 : 클래스 메서드 : .classList.toggle()

이미지를 클릭하면 클래스(active)를 추가 /삭제합니다. 
{
    //모든 이미지 선택  
    document.querySelectorAll(".ex2 div").forEach((el) => {
        el.addEventListener("click", function(){
            document.querySelectorAll(".ex2 div").forEach((el)=> {
                el.classList.toggle("active");
            })
        });
    })    

    // 내가 클릭한 이미지만 선택
    //화살표 함수는 this를 인식하지 못해서 function(){}으로 써야한다.
    document.querySelectorAll(".ex2 div").forEach((el)=>{
        el.addEventListener("click", function (){
            this.classList.toggle("active");
        });  
    })
}

03. 요소객체 : 클래스 메서드 : .classList.contains()

리셋 클릭하면 .active 클래스를 찾은 후에 클래스를 제거합니다. 
{

    document.querySelector(".btn3-1").addEventListener("click",function(){
        document.querySelectorAll(".ex3 > div:nth-child(1)").classList.add("active");
        document.querySelectorAll(".ex3 > div:nth-child(4)").classList.add("active");
    })
        
    document.querySelector(".btn3-1").addEventListener("click", () => {
        document.querySelectorAll(".ex3 > div").forEach(el => {
            if (el.classList.contains("active")) {
                el.classList.remove("active");
            }
        })
    });
