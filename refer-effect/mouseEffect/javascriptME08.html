window.addEventListener("mousemove", function (e) {
    const w = document.querySelector('.cursor').clientWidth / 2;
    const y = document.querySelector('.cursor').clientHeight / 2;
    const mouseX = e.clientX;
    const mouseY = e.clientY;

    const items = document.querySelectorAll(".item");
    items.forEach((item, index) => {
        const itemX = item.offsetLeft + 25;
        const itemY = item.offsetTop + 25;
        const diffX = mouseX - itemX;
        const diffY = mouseY - itemY;
        const radians = Math.atan2(diffY, diffX);
        const angle = radians * 180 / Math.PI;

        item.style.transform = "rotate("+angle+"deg)";
    });


    //커서
    gsap.to('.cursor', {
        duration: .5,
        left: e.pageX - w,
        top: e.pageY - y
    });
});
