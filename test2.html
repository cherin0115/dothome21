{        
     let stuff = prompt('제일 먼저 잡고 싶은 물건은 무엇인가요?');
        switch(stuff){
            case "1.과일":
                alert("당신은 생각이 많고 진지해 보이는 연상 타입입니다.");
                break;
            case "2.떡":
                alert("당신은 무드에 약해 화술과 매너가 좋은 타입입니다.");
                break;
            case "3.수정구슬":
                alert("당신은 정열적이고 열중하는 타입입니다.");
                break;
            case "4.현찰":
                alert("당신은 부드럽고 따듯한 타입입니다.");
                break;
            case "5.반지":
                alert("당신은 자기를 보호해 줄 수 있는 이성 타입입니다.");
                break;
            default:
                alert("입력값이 없습니다.")               
        }
    }
